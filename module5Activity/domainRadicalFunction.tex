\begin{sagesilent}
a1 = ZZ.random_element(2, 10)
\end{sagesilent}

\begin{question}
Write the domain of the function $\sqrt{x - \sage{a1} }$ in interval notation.

$\answer[format=string]{[} \answer{\sage{a1}}, \answer{\sage{Infinity}} \answer[format=string]{)}$

\begin{feedback}

\end{feedback}

\end{question}
